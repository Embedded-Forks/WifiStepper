# ============= EXAMPLE CODE =============
# Uses curl for network and jq for json parsing
HOST="{{ service.hostname }}.local"  # or "{{ wifi.ip }}"
PROTOCOL="http"  # or https

# Ping motor api
curl -sS "${PROTOCOL}://${HOST}/api/ping" | jq
# All successful api calls will return 'status: ok'. This means
#  that the command was successfully interpreted and executed.
#  However, errors may still be thrown during execution.

# Write motor configuration
MODE="{{ mode }}"
STEPSIZE="{{ stepsize }}"
OCD="{{ ocd }}"
OCDSHUTDOWN="{{ ocdshutdown }}"
MAXSPEED="{{ maxspeed }}"
MINSPEED="{{ minspeed }}"
ACCEL="{{ accel }}"
DECEL="{{ decel }}"
CM_KTHOLD="{{ cm_kthold }}"
CM_KTRUN="{{ cm_ktrun }}"
CM_KTACCEL="{{ cm_ktaccel }}"
CM_KTDECEL="{{ cm_ktdecel }}"
VM_KTHOLD="{{ vm_kthold }}"
VM_KTRUN="{{ vm_ktrun }}"
VM_KTACCEL="{{ vm_ktaccel }}"
VM_KTDECEL="{{ vm_ktdecel }}"
FSSPEED="{{ fsspeed }}"
FSBOOST="{{ fsboost }}"
REVERSE="{{ reverse }}"
SAVE_EEPROM="{{ save }}"

# Advanced configuration settings
CM_SWITCHPERIOD="{{ cm_switchperiod }}"
CM_PREDICT="{{ cm_predict }}"
CM_MINON="{{ cm_minon }}"
CM_MINOFF="{{ cm_minoff }}"
CM_FASTOFF="{{ cm_fastoff }}"
CM_FASTSTEP="{{ cm_faststep }}"
VM_PWMFREQ="{{ vm_pwmfreq }}"
VM_STALL="{{ vm_stall }}"
VM_BMEF_SLOPEL="{{ vm_bemf_slopel }}"
VM_BMEF_SPEEDCO="{{ vm_bemf_speedco }}"
VM_BMEF_SLOPEHACC="{{ vm_bemf_slopehacc }}"
VM_BMEF_SLOPEHDEC="{{ vm_bemf_slopehdec }}"

curl -sS "${PROTOCOL}://${HOST}/api/motor/set?\
mode=${MODE}&stepsize=${STEPSIZE}&\
ocd=${OCD}&ocdshutdown=${OCDSHUTDOWN}&\
maxspeed=${MAXSPEED}&minspeed=${MINSPEED}&accel=${ACCEL}&decel=${DECEL}&\
cm_kthold=${CM_KTHOLD}&cm_ktrun=${CM_KTRUN}&cm_ktaccel=${CM_KTACCEL}&cm_ktdecel=${CM_KTDECEL}&\
vm_kthold=${VM_KTHOLD}&vm_ktrun=${VM_KTRUN}&vm_ktaccel=${VM_KTACCEL}&vm_ktdecel=${VM_KTDECEL}&\
fsspeed=${FSSPEED}&fsboost=${FSBOOST}&\
cm_switchperiod=${CM_SWITCHPERIOD}&cm_predict=${CM_PREDICT}&\
cm_minon=${CM_MINON}&cm_minoff=${CM_MINOFF}&cm_fastoff=${CM_FASTOFF}&cm_faststep=${CM_FASTSTEP}&\
vm_pwmfreq=${VM_PWMFREQ}&vm_stall=${VM_STALL}&\
vm_bmef_slopel=${VM_BMEF_SLOPEL}&vm_bmef_speedco=${VM_BMEF_SPEEDCO}&\
vm_bmef_slopehacc=${VM_BMEF_SLOPEHACC}&vm_bmef_slopehdec=${VM_BMEF_SLOPEHDEC}&\
reverse=${REVERSE}&save=${SAVE_EEPROM}" | jq

# Read motor configuration
curl -sS "${PROTOCOL}://${HOST}/api/motor/get" | jq
# Returns object with configuration map

# Zero position
curl -sS "${PROTOCOL}://${HOST}/api/motor/pos/reset" | jq

# Send servo command
POSITION="{{ setpos }}"  # ANGLE / 360 * 200 * STEPSIZE; ANGLE={{ setangle }}, STEPSIZE={{ stepsize }}
curl -sS "${PROTOCOL}://${HOST}/api/motor/goto?position=${POSITION}" | jq

# Send speed command
STEPS_PER_SEC="{{ setstepss }}"  # RPM / 60 * 200; RPM={{ setrpm }}
DIRECTION="{{ setdirection }}"
curl -sS "${PROTOCOL}://${HOST}/api/motor/run?stepss=${STEPS_PER_SEC}&direction=${DIRECTION}" | jq

# Send stepclock command
DIRECTION="{{ setdirection }}"
curl -sS "${PROTOCOL}://${HOST}/api/motor/stepclock?direction=${DIRECTION}" | jq

# Stop
STOP_HIZ="{{ speed.hiz }}"
STOP_SOFT="true"
curl -sS "${PROTOCOL}://${HOST}/api/motor/stop?hiz=${STOP_HIZ}&soft=${STOP_SOFT}" | jq

# Read state
curl -sS "${PROTOCOL}://${HOST}/api/motor/state" | jq
# Returns object with keys: stepss, position, mark, adc, direction, movement, hiz, busy, switch, stepclock,
#   flags:{ switch, commanderror, overcurrent, undervoltage, thermalwarning,
#   thermalshutdown, stalldetect }
